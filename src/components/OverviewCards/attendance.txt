export default function AttendanceOverviewCard({
  setActive,
  data,
  loading = false,
  error = null
}) {
  if (loading) {
    return <div className="overview-card">Loading attendance...</div>;
  }

  if (error) {
    return <div className="overview-card error">Failed to load data</div>;
  }

  if (!data) return null;

  return (
    <div className="overview-card">
      <h3
        onClick={() => setActive("Attendance")}
        className="card-title clickable"
      >
        Attendance Overview
      </h3>

      <div className="card-body">
        <p>
          <strong>Department Average:</strong>{" "}
          <span className={getAttendanceStatus(data.departmentAvg)}>
            {data.departmentAvg}%
          </span>
        </p>

        <PercentageBar value={data.departmentAvg} />

        <div className="card-section">
          {data.classes.map(cls => (
            <div key={cls.year} className="class-row">
              <span>{cls.year}</span>
              <span className={getAttendanceStatus(cls.avg)}>
                {cls.avg}%
              </span>
              <PercentageBar value={cls.avg} />
            </div>
          ))}
        </div>

        <p className="subtle">
          Click to view detailed attendance records
        </p>
      </div>
    </div>
  );
}
